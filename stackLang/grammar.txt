Program -> Stmts Returnst
Stmts -> Stmt Stmts_
Stmt -> Splitst | Whilest | Ifst | Op
Splitst -> split numberLiteral* Op_ . Op_ . Splitclose .
Splitclose -> swap | stack | merge
Whilest -> while Signal Stmts .
Ifst -> if Signal Stmts . Elsest_
Elsest -> else Stmts .
Returnst -> return
Op -> ArithmeticOp | push PushOp | pop PopOp | rotate | dump IO
IO -> in | out
PushOp -> numberLiteral* | @ SignalName | input numberLiteral*
PopOp -> numberLiteral* | @ SignalName | output numberLiteral*
ArithmeticOp -> add | sub | neg | and | not | or | gt | lt | eq | swap | SymbolicOp
SymbolicOp -> + | - | ~ | & | ! | v | > | < | = | x
Signal -> numberLiteral* | @ SignalName
SignalName -> KeySource
KeySource -> height | top | temp
